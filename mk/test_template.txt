$(MODULE_NAME)_test_path_curdir				        := $(dir $(lastword $(MAKEFILE_LIST)))
$(MODULE_NAME)_test_child_makefiles			        := $(wildcard $($(MODULE_NAME)_test_path_curdir)*/*mk)
$(MODULE_NAME)_test_child_module_names		        := $(basename $(notdir $($(MODULE_NAME)_test_child_makefiles)))
$(MODULE_NAME)_test_child_all_targets		        := $(foreach test_module,$($(MODULE_NAME)_test_child_module_names),$(test_module)_all)
$(MODULE_NAME)_test_child_clean_targets		        := $(foreach test_module,$($(MODULE_NAME)_test_child_module_names),$(test_module)_clean)
$(MODULE_NAME)_test_child_run_targets		        := $(foreach test_module,$($(MODULE_NAME)_test_child_module_names),$(test_module)_run)
$(MODULE_NAME)_test_install_path_static		        := $($(MODULE_NAME)_test_path_curdir)$(MODULE_NAME)_static$(EXT_EXE)
$(MODULE_NAME)_test_sources					        := $(wildcard $($(MODULE_NAME)_test_path_curdir)*.c)
$(MODULE_NAME)_test_objects					        := $(patsubst %.c, %.o, $($(MODULE_NAME)_test_sources))
$(MODULE_NAME)_test_depends					        := $(patsubst %.c, %.d, $($(MODULE_NAME)_test_sources))
$(MODULE_NAME)_test_depends_modules			        := $(MODULE_LIBDEP_MODULES)
# $(MODULE_NAME)_test_depends_modules			        += test_framework
$(MODULE_NAME)_test_depends_modules			        += $(MODULE_NAME)
$(MODULE_NAME)_test_libdepend_static_objs	        := $(foreach dep_module,$($(MODULE_NAME)_depends_modules),$($(dep_module)_static_objects))
$(MODULE_NAME)_test_libdepend_static_objs	        += $(foreach dep_module,$(foreach m,$($(MODULE_NAME)_test_depends_modules),$($(m)_depends_modules)),$($(dep_module)_static_objects))
$(MODULE_NAME)_test_libdepend_static_objs	        += $(foreach dep_module,$($(MODULE_NAME)_test_depends_modules),$($(dep_module)_static_objects))

include $($(MODULE_NAME)_test_child_makefiles)

$($(MODULE_NAME)_test_path_curdir)%.o: $($(MODULE_NAME)_test_path_curdir)%.c
	$(CC) -c $< -o $@ $(CFLAGS_COMMON) -MMD -MP -MF $(<:.c=.d) -DGIL_LIB_SHARED_EXPORT

$($(MODULE_NAME)_test_install_path_static): $($(MODULE_NAME)_test_libdepend_static_objs)
$($(MODULE_NAME)_test_install_path_static): $($(MODULE_NAME)_test_objects)
	$(CC) -o $@ $($(MODULE_NAME)_test_objects) -Wl,--allow-multiple-definition $($(MODULE_NAME)_test_libdepend_static_objs) $(LFLAGS_COMMON) $(LFLAGS_SPECIFIC)

.PHONY: $(MODULE_NAME)_test_all
$(MODULE_NAME)_test_all: $($(MODULE_NAME)_test_child_all_targets) ## build all $(MODULE_NAME)_test tests
ifneq ($($(MODULE_NAME)_test_objects),)
$(MODULE_NAME)_test_all: $($(MODULE_NAME)_test_install_path_static)
endif

.PHONY: $(MODULE_NAME)_test_clean
$(MODULE_NAME)_test_clean: $($(MODULE_NAME)_test_child_clean_targets) ## remove all $(MODULE_NAME)_test tests
$(MODULE_NAME)_test_clean:
	- $(RM) $($(MODULE_NAME)_test_install_path_static) $($(MODULE_NAME)_test_objects) $($(MODULE_NAME)_test_depends)

.PHONY: $(MODULE_NAME)_test_re
$(MODULE_NAME)_test_re: $(MODULE_NAME)_test_clean
$(MODULE_NAME)_test_re: $(MODULE_NAME)_test_all

.PHONY: $(MODULE_NAME)_test_run_all
$(MODULE_NAME)_test_run_all: $(MODULE_NAME)_test_all ## build and run static $(MODULE_NAME)_test
$(MODULE_NAME)_test_run_all: $($(MODULE_NAME)_test_child_run_targets)
ifneq ($($(MODULE_NAME)_test_objects),)
$(MODULE_NAME)_test_run_all: $(PATH_INSTALL)/test_framework$(EXT_EXE)
	@$(PATH_INSTALL)/test_framework$(EXT_EXE) $($(MODULE_NAME)_test_install_path_static)
#	@$(PYTHON) $(PATH_MK_FILES)/pytester.py $($(MODULE_NAME)_test_install_path_static)
endif

.PHONY: $(MODULE_NAME)_test_run
$(MODULE_NAME)_test_run: $(MODULE_NAME)_test_all
ifneq ($($(MODULE_NAME)_test_objects),)
$(MODULE_NAME)_test_run: $(PATH_INSTALL)/test_framework$(EXT_EXE)
	@$(PATH_INSTALL)/test_framework$(EXT_EXE) $($(MODULE_NAME)_test_install_path_static)
#	@$(PYTHON) $(PATH_MK_FILES)/pytester.py $($(MODULE_NAME)_test_install_path_static)
endif
